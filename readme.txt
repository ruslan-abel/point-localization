	Vertices(Вершины) 			вектор Point
	Edge(Ребро) 				пара из индексов точек
	Edges(Рёбра) 				вектор Edge
	Edge_List(РСДС)				рёберный список с двойными связями
	PPLG(Плоский планарный граф)		основная структура
